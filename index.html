async function sync() { const input = document.getElementById('userInput'); const text = input.value.trim(); if(!text) return; let isNew = false; if(activeId === null) { activeId = Date.now(); chats.unshift({ id: activeId, name: "Neural Sync...", history: [] }); isNew = true; } const chat = chats.find(c => c.id === activeId); addUI(text, 'user'); chat.history.push({ role: 'user', text }); input.value = ''; document.getElementById('status').innerText = "LINKED / EXTRACTING..."; try { const res = await fetch('/api/sync', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ prompt: text, history: chat.history, globalMemory: globalMemory, // Sending everything he knows so far generateTitle: isNew }) }); const data = await res.json(); // AUTO-UPDATE GLOBAL MEMORY // If the AI returns a 'memoryUpdate', we save it permanently if (data.memoryUpdate) { globalMemory += ` [Detail: ${data.memoryUpdate}]`; save(); // Persists to localStorage immediately } if(isNew && data.title) chat.name = data.title; addUI(data.text, 'twin'); chat.history.push({ role: 'twin', text: data.text }); document.getElementById('status').innerText = "CONNECTED"; save(); } catch (e) { addUI("Link Severed.", 'twin'); } }
